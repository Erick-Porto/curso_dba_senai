Exercício 3 - Solução Comentada

Query SQL:


-- Seleciona valores distintos da coluna Cidade, renomeando-a
SELECT DISTINCT
    Cidade AS CidadeAtendida
-- Da tabela Clientes
FROM
    Clientes
-- Onde a data de cadastro é anterior a 01/01/2023
WHERE
    DataCadastro < '2023-01-01'
    -- E a idade é menor que 30
    AND Idade < 30
-- Ordena o resultado pelo nome da cidade (A-Z), usando o alias
ORDER BY
    CidadeAtendida ASC; -- Ou ORDER BY Cidade ASC;
