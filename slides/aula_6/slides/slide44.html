-- Solução para Exercício 2: Criação da Tabela Produtos e Consultas

-- Tarefa 1: Criação da Tabela Produtos
CREATE TABLE Produtos (
    ProdutoID INT PRIMARY KEY AUTO_INCREMENT, -- MySQL
    -- ProdutoID SERIAL PRIMARY KEY, -- PostgreSQL
    NomeProduto VARCHAR(100) NOT NULL,
    Preco DECIMAL(10, 2) NOT NULL CHECK (Preco > 0), -- Preço com 2 casas decimais, maior que 0

    -- Chave estrangeira referenciando a tabela Categorias
    CategoriaID INT,
    FOREIGN KEY (CategoriaID) REFERENCES Categorias(CategoriaID)
      -- ON DELETE SET NULL -- Opcional: O que fazer se a categoria for deletada?
      -- ON UPDATE CASCADE -- Opcional: O que fazer se o CategoriaID mudar?
);

INSERT INTO Produtos (NomeProduto, Preco, CategoriaID) VALUES

-- Tarefa 3: Consulta de Produtos com Preço > 50
SELECT NomeProduto, Preco
FROM Produtos
WHERE Preco > 50.00;
