Solução: Exercício 1 - sp_BuscarProdutosPorFaixaDePreco


DELIMITER $$
CREATE PROCEDURE sp_BuscarProdutosPorFaixaDePreco (
    IN preco_minimo DECIMAL(10,2),
    IN preco_maximo DECIMAL(10,2)
)
BEGIN
    -- Seleciona os produtos dentro da faixa de preço especificada
    SELECT IDProduto, Nome, Preco, Estoque
    FROM Produtos
    WHERE Preco >= preco_minimo AND Preco <= preco_maximo
    ORDER BY Preco ASC;
END $$
DELIMITER ;

-- Exemplo de chamada:
-- CALL sp_BuscarProdutosPorFaixaDePreco(10.00, 50.00);

Comentários: A SP recebe os limites de preço, filtra os produtos usando WHERE e ordena com ORDER BY.
